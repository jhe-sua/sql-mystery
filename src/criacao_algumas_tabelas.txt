
CREATE USER investigador_comum WITH PASSWORD 'senha_investigador';
CREATE USER investigador_autorizado WITH PASSWORD 'senha_autorizada';


GRANT SELECT, INSERT, UPDATE ON falecidos, contas_bancarias, saques, 
      declaracao_renda, suspeitos, movimentos_suspeitos TO investigador_comum;


GRANT SELECT, INSERT, UPDATE ON tabela_proibida_identidades TO investigador_autorizado;


REVOKE ALL ON tabela_proibida_identidades FROM investigador_comum;






-- 1. Tabela de pessoas falecidas
CREATE TABLE falecidos (
    id_falecido SERIAL PRIMARY KEY,
    nome_completo VARCHAR(200) NOT NULL,
    cpf VARCHAR(14) UNIQUE NOT NULL,
    data_obito DATE NOT NULL,
    certificado_obito VARCHAR(50) UNIQUE NOT NULL
);

-- 2. Tabela de contas bancárias 
CREATE TABLE contas_bancarias (
    id_conta SERIAL PRIMARY KEY,
    cpf_titular VARCHAR(14) NOT NULL,
    banco VARCHAR(100),
    agencia VARCHAR(20),
    numero_conta VARCHAR(30) UNIQUE,
    saldo DECIMAL(12,2)
);

-- 3. Tabela de saques
CREATE TABLE saques (
    id_saque SERIAL PRIMARY KEY,
    id_conta INTEGER REFERENCES contas_bancarias(id_conta),
    valor DECIMAL(10,2) NOT NULL,
    data_saque TIMESTAMP NOT NULL,
    cpf_quem_sacou_encriptografados VARCHAR(14) NOT NULL, -- CPF de quem realizou o saque
    localizacao VARCHAR(255)
);

-- 4. Tabela de declaração de renda (anual)
CREATE TABLE declaracao_renda (
    id_declaracao SERIAL PRIMARY KEY,
    cpf VARCHAR(14) NOT NULL,
    ano_referencia INTEGER NOT NULL,
    renda_declarada DECIMAL(12,2),
    renda_real_estimada DECIMAL(12,2), -- para inconsistências
    origem_principal VARCHAR(100)
);

-- 5. Tabela de possíveis suspeitos (vivos)
CREATE TABLE suspeitos (
    id_vivo SERIAL PRIMARY KEY,
    cpf VARCHAR(14) UNIQUE NOT NULL,
    nome VARCHAR(200),
    ultima_renda_declarada money,
    renda_anterior money,
);

-- 6. Tabela proibida (acesso restrito na investigação)
CREATE TABLE tabela_proibida_identidades (
    id_proibido SERIAL PRIMARY KEY,
    cpf_quem_sacou_encriptografados VARCHAR(14),
    cpf VARCHAR(14) UNIQUE NOT NULL,
    nome VARCHAR(200),
    data_nascimento DATE    
FOREIGN KEY (cpf_quem_sacou_encriptografados) REFERENCES saques(cpf_quem_sacou_encriptografados)
);

-- 7. Movimentações suspeitas (viagens, reconhecimento facial trocado por senha etc)
CREATE TABLE movimentos_suspeitos (
    id_movimento SERIAL PRIMARY KEY,
    cpf_suspeito VARCHAR(14) REFERENCES suspeitos(cpf),
    descricao TEXT,
    data_ocorrencia DATE,
    tipo VARCHAR(50) -- 'viagem cara', 'troca_senha_facial', 'testemunha'
);







-- 8. Log do jogador (para ativar o "banco se apagar")
CREATE TABLE controle_jogo (
    chave VARCHAR(50) PRIMARY KEY,
    valor_texto TEXT,
    valor_alerta BOOLEAN DEFAULT FALSE
);



CREATE OR REPLACE FUNCTION apagar_banco_se_culpado_descoberto()
RETURNS TRIGGER AS $$
DECLARE
    celula_controle TEXT;
BEGIN
    -- Verifica se a célula de controle (ex: 'verdadeiro_culpado' tem 'REVELADO')
    SELECT valor_texto INTO celula_controle FROM controle_jogo WHERE chave = 'gatilho_apagar';
    
    IF celula_controle = 'REVELADO' THEN
        TRUNCATE falecidos, contas_bancarias, saques, declaracao_renda, 
                 suspeitos, movimentos_suspeitos RESTART IDENTITY CASCADE;
    END IF;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Gatilho que monitora mudanças na tabela controle_jogo
CREATE TRIGGER vigilancia_revelacao
AFTER UPDATE ON controle_jogo
FOR EACH ROW
WHEN (NEW.chave = 'gatilho_apagar' AND NEW.valor_texto = 'REVELADO')
EXECUTE FUNCTION apagar_banco_se_culpado_descoberto();
